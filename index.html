<script>

const canvas = document.getElementById('c');

const ctx = canvas.getContext('2d');

const W = canvas.width, H = canvas.height;


let bodies = [], paused = false, showTrail = true, G = 600, speedMult = 1;

const COLORS = ['#4af','#fa4','#f4a','#4fa','#a4f','#f44','#4ff','#ff4','#af4'];

const TRAIL_LEN = 200;


function makeStar(x, y, vx, vy, mass, color) {

  return { x, y, vx, vy, mass, color, trail: [], r: Math.cbrt(mass) * 1.5 };

}


function spawnDefault() {

  bodies = [];

  const cx = W/2, cy = H/2;

  bodies.push(makeStar(cx, cy, 0, 0, 9000, '#ffe08a'));

  const orbits = [

    {r:90,  m:120, c:'#4af'},

    {r:145, m:200, c:'#4fa'},

    {r:205, m:80,  c:'#f4a'},

    {r:270, m:300, c:'#a4f'},

  ];

  orbits.forEach(o => {

    const angle = Math.random() * Math.PI * 2;

    const x = cx + Math.cos(angle) * o.r;

    const y = cy + Math.sin(angle) * o.r;

    const speed = Math.sqrt(G * 9000 / o.r);

    bodies.push(makeStar(x, y, -Math.sin(angle)*speed, Math.cos(angle)*speed, o.m, o.c));

  });

}

spawnDefault();


let bgStars = [];

for(let i = 0; i < 200; i++) bgStars.push({x: Math.random()*W, y: Math.random()*H, b: Math.random()});


let lastTime = null;

function step(ts) {

  if (!lastTime) lastTime = ts;

  let dt = Math.min((ts - lastTime) / 1000, 0.03) * speedMult;

  lastTime = ts;


  if (!paused) {

    for (let i = 0; i < bodies.length; i++) {

      bodies[i].ax = 0; bodies[i].ay = 0;

      for (let j = 0; j < bodies.length; j++) {

        if (i === j) continue;

        const dx = bodies[j].x - bodies[i].x;

        const dy = bodies[j].y - bodies[i].y;

        const d2 = dx*dx + dy*dy;

        const d = Math.sqrt(d2);

        if (d < bodies[i].r + bodies[j].r) continue;

        const f = G * bodies[j].mass / (d2 + 80);

        bodies[i].ax += f * dx / d;

        bodies[i].ay += f * dy / d;

      }

    }

    bodies.forEach(b => {

      b.vx += b.ax * dt;

      b.vy += b.ay * dt;

      b.x  += b.vx * dt;

      b.y  += b.vy * dt;

      b.trail.push({x: b.x, y: b.y});

      if (b.trail.length > TRAIL_LEN) b.trail.shift();

    });

    bodies = bodies.filter((b, i) => i === 0 || (b.x > -W && b.x < W*2 && b.y > -H && b.y < H*2));

    document.getElementById('nb').textContent = bodies.length;

  }


  // Clear

  if (showTrail) {

    ctx.fillStyle = 'rgba(2,6,16,0.16)';

  } else {

    ctx.fillStyle = '#020610';

  }

  ctx.fillRect(0, 0, W, H);


  // BG stars

  bgStars.forEach(s => {

    const a = 0.15 + 0.25 * Math.sin(ts * 0.0008 + s.b * 9);

    ctx.fillStyle = `rgba(190,210,255,${a})`;

    ctx.beginPath(); ctx.arc(s.x, s.y, 0.8, 0, Math.PI*2); ctx.fill();

  });


  // Trails

  if (showTrail) {

    bodies.forEach(b => {

      if (b.trail.length < 2) return;

      for (let i = 1; i < b.trail.length; i++) {

        const a = (i / b.trail.length) * 0.75;

        ctx.beginPath();

        ctx.strokeStyle = b.color + Math.floor(a * 255).toString(16).padStart(2, '0');

        ctx.lineWidth = 0.7 + a * 1.5;

        ctx.moveTo(b.trail[i-1].x, b.trail[i-1].y);

        ctx.lineTo(b.trail[i].x,   b.trail[i].y);

        ctx.stroke();

      }

    });

  }


  // Bodies

  bodies.forEach(b => {

    // Glow

    for (let g = 4; g > 0; g--) {

      ctx.beginPath();

      ctx.arc(b.x, b.y, b.r + g * 3.5, 0, Math.PI*2);

      ctx.fillStyle = b.color + ['18','11','09','05'][g-1];

      ctx.fill();

    }

    ctx.beginPath();

    ctx.arc(b.x, b.y, b.r, 0, Math.PI*2);

    ctx.fillStyle = b.color;

    ctx.fill();

    // Specular

    ctx.beginPath();

    ctx.arc(b.x - b.r * 0.3, b.y - b.r * 0.35, b.r * 0.38, 0, Math.PI*2);

    ctx.fillStyle = 'rgba(255,255,255,0.3)';

    ctx.fill();

    // Mass label

    if (b.mass > 500) {

      ctx.fillStyle = 'rgba(255,240,150,0.6)';

      ctx.font = '10px monospace';

      ctx.textAlign = 'center';

      ctx.fillText('M=' + Math.round(b.mass), b.x, b.y - b.r - 5);

    }

  });


  requestAnimationFrame(step);

}

requestAnimationFrame(step);


canvas.addEventListener('click', e => {

  const rect = canvas.getBoundingClientRect();

  const mx = (e.clientX - rect.left) * (W / rect.width);

  const my = (e.clientY - rect.top)  * (H / rect.height);

  let nearest = bodies[0], minD = Infinity;

  bodies.forEach(b => { const d = Math.hypot(b.x-mx, b.y-my); if(d<minD){minD=d;nearest=b;} });

  const dx = nearest.x - mx, dy = nearest.y - my;

  const d = Math.hypot(dx, dy) || 1;

  const speed = Math.sqrt(G * nearest.mass / d) * 0.88;

  const col = COLORS[Math.floor(Math.random() * COLORS.length)];

  const mass = 60 + Math.random() * 200;

  bodies.push(makeStar(mx, my, -(dy/d)*speed + nearest.vx, (dx/d)*speed + nearest.vy, mass, col));

});


document.getElementById('btn-pause').onclick = function() {

  paused = !paused; lastTime = null;

  this.textContent = paused ? '▶ Play' : '⏸ Pause';

};

document.getElementById('btn-reset').onclick = () => spawnDefault();

document.getElementById('btn-trail').onclick = function() {

  showTrail = !showTrail;

  this.textContent = 'Trails: ' + (showTrail ? 'ON' : 'OFF');

};

document.getElementById('speed').oninput = function() {

  speedMult = parseFloat(this.value);

  document.getElementById('spv').textContent = speedMult.toFixed(1) + '×';

};

document.getElementById('grav').oninput = function() {

  G = parseInt(this.value);

  document.getElementById('gv').textContent = G;

};

</script>
